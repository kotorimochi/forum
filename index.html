function displayData(records) { const dataContainer = document.getElementById('data'); dataContainer.innerHTML = ''; // データの初期化 records.forEach((record, index) => { const recordElement = document.createElement('div'); recordElement.className = 'record'; recordElement.textContent = `レコード ${index + 1}`; const detailElement = document.createElement('div'); detailElement.className = 'record-detail'; Object.keys(record).forEach(fieldKey => { // 表示したくないフィールドをスキップ const fieldsToSkip = ["更新者", "作成者", "更新日時", "作成日時"]; if (fieldsToSkip.includes(fieldKey)) { return; } const fieldValue = record[fieldKey].value; // リッチテキストフィールドの処理 if (fieldKey === "内容_リッチ") { const richFieldElement = document.createElement('div'); richFieldElement.innerHTML = `${fieldKey}: ${fieldValue}`; detailElement.appendChild(richFieldElement); } else { const fieldElement = document.createElement('p'); fieldElement.textContent = `${fieldKey}: ${fieldValue}`; detailElement.appendChild(fieldElement); } }); // クリックで詳細表示/非表示を切り替え recordElement.addEventListener('click', () => { detailElement.style.display = detailElement.style.display === 'none' ? 'block' : 'none'; }); // レコードと詳細を追加 dataContainer.appendChild(recordElement); dataContainer.appendChild(detailElement); }); }
